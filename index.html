// Daniel's CEO Dashboard - Calendar Sync Script // Pulls Exercise calendar events from the past month, writes each as a row to the Sync sheet // Run on a schedule: Apps Script > Triggers > New trigger > Time-driven > Daily > 00:00 function syncCalendarToSheet() { // Config var CALENDAR_ID = "e29a920cea47dd69a4b033d18485ecaa82ade6d76f321b439a60e6325b5d7dc7@group.calendar.google.com"; var SHEET_ID = "1b-Smzvo1066R2mH4uIZ7nPlNIYscqlU8EarSAhXo2Xw"; var SHEET_NAME = "Sync"; // Open the sheet and get or create the Sync tab var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } // Set up headers if empty if (sheet.getLastRow() === 0) { sheet.appendRow(["date", "event_name"]); } // Fetch calendar events from the past month var now = new Date(); var monthAgo = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000); var calendar = CalendarApp.getCalendarById(CALENDAR_ID); if (!calendar) { Logger.log("Calendar not found. Check CALENDAR_ID."); return; } var events = calendar.getEvents(monthAgo, now); // Clear old data (keep header) if (sheet.getLastRow() > 1) { sheet.deleteRows(2, sheet.getLastRow() - 1); } // Write each event as a row var rows = []; events.forEach(function(event) { if (!event.isAllDayEvent()) { var eventDate = Utilities.formatDate(event.getStartTime(), Session.getScriptTimeZone(), "yyyy-MM-dd"); var eventName = event.getTitle(); rows.push([eventDate, eventName]); } }); // Sort by date descending (most recent first) rows.sort(function(a, b) { return new Date(b[0]) - new Date(a[0]); }); // Append all rows at once if (rows.length > 0) { sheet.getRange(2, 1, rows.length, 2).setValues(rows); } Logger.log("Synced " + rows.length + " calendar events to " + SHEET_NAME); } // Run this once to set up the daily trigger function setupTrigger() { // Delete any existing triggers for this function var triggers = ScriptApp.getProjectTriggers(); triggers.forEach(function(trigger) { if (trigger.getHandlerFunction() === "syncCalendarToSheet") { ScriptApp.deleteTrigger(trigger); } }); // Create new daily trigger at 00:30 UTC ScriptApp.newTrigger("syncCalendarToSheet") .timeBased() .atHour(0) .nearMinute(30) .everyDays(1) .create(); Logger.log("Trigger set up. Will run daily at 00:30."); }
